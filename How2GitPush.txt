如果本地已经有af-qmc目录
1. 初始化目录为git仓库
cd /mnt/f/af-qmc
git init  (first time for this repository)
git remote add origin https://github.com/jinyd1983/af-qmc.git

2. 文件添加到仓库中
git add af-qmc_simple.py

3. 提交更改
git commit -m "add af-qmc_simple.py"

4. 推送到远程main分支
git push origin main




#如果本地没有af-qmc目录
# 1. 克隆仓库到本地（如果还没有克隆）
git clone https://github.com/jinyd/af-qmc.git
cd af-qmc

# 2. 将af-qmc_simple.py文件添加到仓库中
git add af-qmc_simple.py

# 3. 提交更改
git commit -m "添加简单的AF-QMC Python实现"

# 4. 推送到远程main分支
git push origin main
